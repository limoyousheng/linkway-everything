```js
function switchMobileView(view) {
    state.view = view;
    if (view === 'hub' || view === 'archive') {
        state.currentCategoryId = null;
    }
    renderAll();
    updateMobileTabUI();
    if (isMobile()) scrollToTop();
}

function updateMobileTabUI() {
    const tabs = document.querySelectorAll('#mobile-bottom-nav .mob-tab');
    tabs.forEach(tab => {
        const v = tab.dataset.tab;
        const isActive = v === state.view;
        const indicator = tab.querySelector('.mob-tab-indicator');
        const icon = tab.querySelector('.mob-tab-icon');
        const label = tab.querySelector('.mob-tab-label');
        if (isActive) {
            indicator.classList.remove('bg-transparent');
            indicator.classList.add('bg-primary');
            icon.style.fontVariationSettings = "'FILL' 1";
            label.classList.add('text-primary', 'font-semibold');
            label.classList.remove('text-on-surface-variant');
        } else {
            indicator.classList.add('bg-transparent');
            indicator.classList.remove('bg-primary');
            icon.style.fontVariationSettings = "'FILL' 0";
            label.classList.remove('text-primary', 'font-semibold');
            label.classList.add('text-on-surface-variant');
        }
    });
}

function scrollToTop() {
    window.scrollTo({ top: 0, behavior: isMobile() ? 'smooth' : 'auto' });
}
```

function bindSwipeToActions(cardEl, cardId) {
    if (!cardEl) return;
    let startX = 0, startY = 0, startTime = 0, swiping = false;
    const SWIPE_THRESHOLD = 50;  // 触发阈值

    cardEl.addEventListener('touchstart', (e) => {
        if (e.touches.length !== 1) return;
        startX = e.touches[0].clientX;
        startY = e.touches[0].clientY;
        startTime = Date.now();
        swiping = false;
    }, { passive: true });

    cardEl.addEventListener('touchmove', (e) => {
        if (e.touches.length !== 1) return;
        const dx = e.touches[0].clientX - startX;
        const dy = e.touches[0].clientY - startY;
        if (Math.abs(dy) > Math.abs(dx) * 1.2) return;  // 竖滑不触发
        if (dx < -SWIPE_THRESHOLD && !swiping) {
            swiping = true;
            if (navigator.vibrate) navigator.vibrate(10);
            openActionSheet(cardId);
        }
    }, { passive: true });
}

function showBottomSheet({ content, height = 'auto', dismissible = true, onMount, onDismiss } = {}) {
    const root = document.getElementById('mobile-sheet-root');
    if (!root) return null;
    // 关闭已有 sheet
    root.innerHTML = '';
    root.classList.remove('hidden');

    const sheet = document.createElement('div');
    sheet.className = 'mob-sheet fixed bottom-0 inset-x-0 z-50 bg-surface-container-low rounded-t-3xl shadow-2xl flex flex-col max-h-[92vh]';
    sheet.style.paddingBottom = 'var(--safe-bottom)';
    sheet.style.height = height === 'auto' ? 'auto' : height;
    if (height === 'auto') {
        sheet.style.maxHeight = '92vh';
    }

    const handle = document.createElement('div');
    handle.className = 'mob-sheet-handle w-10 h-1 rounded-full bg-on-surface-variant/30 mx-auto my-2.5 shrink-0';
    sheet.appendChild(handle);

    const inner = document.createElement('div');
    inner.className = 'flex-1 overflow-y-auto px-4 pb-4';
    if (typeof content === 'string') inner.innerHTML = content;
    else inner.appendChild(content);
    sheet.appendChild(inner);

    root.appendChild(sheet);

    // 背景遮罩
    const backdrop = document.createElement('div');
    backdrop.className = 'fixed inset-0 z-40 bg-scrim/40';
    backdrop.addEventListener('click', () => dismiss());
    root.appendChild(backdrop);

    // 滑入动画
    requestAnimationFrame(() => {
        sheet.style.transform = 'translateY(0)';
        backdrop.style.opacity = '1';
    });
    sheet.style.transform = 'translateY(100%)';
    sheet.style.transition = 'transform 300ms cubic-bezier(0.16, 1, 0.3, 1)';
    backdrop.style.opacity = '0';
    backdrop.style.transition = 'opacity 200ms ease';

    function dismiss() {
        sheet.style.transform = 'translateY(100%)';
        backdrop.style.opacity = '0';
        setTimeout(() => {
            root.innerHTML = '';
            root.classList.add('hidden');
            if (onDismiss) onDismiss();
        }, 250);
    }

    if (dismissible) {
        // 上滑手势
        let startY = 0, dragging = false;
        handle.addEventListener('touchstart', (e) => {
            startY = e.touches[0].clientY;
            dragging = true;
        }, { passive: true });
        handle.addEventListener('touchmove', (e) => {
            if (!dragging) return;
            const dy = e.touches[0].clientY - startY;
            if (dy > 0) sheet.style.transform = `translateY(${dy}px)`;
        }, { passive: true });
        handle.addEventListener('touchend', (e) => {
            if (!dragging) return;
            dragging = false;
            const dy = (e.changedTouches[0].clientY - startY);
            if (dy > 80) dismiss();
            else sheet.style.transform = 'translateY(0)';
        });
    }

    if (onMount) onMount({ dismiss, sheet, inner });

    return { dismiss, sheet, inner };
}

function closeMobileSheet() {
    const root = document.getElementById('mobile-sheet-root');
    if (root) {
        root.innerHTML = '';
        root.classList.add('hidden');
    }
}

        delay: 75,

        ghostClass: 'sortable-ghost',
        dragClass: 'sortable-drag',

        fallbackClass: 'sortable-drag',

        onEnd: (evt) => {
            document.body.classList.remove('is-dragging');
            // 如果拖到分类网格（onAdd 已在 sidebar handler 里处理），这里只处理网格内重排
            if (evt.from === evt.to && evt.oldIndex !== evt.newIndex) {
                const newOrderIds = Array.from(evt.to.children)
                    .filter(el => el.classList.contains('sortable-item') || el.classList.contains('mob-card'))
                    .map(el => el.dataset.cardId);
                const reordered = newOrderIds.map(id => state.cards.find(c => c.id === id)).filter(Boolean);
                const others = state.cards.filter(c => !newOrderIds.includes(c.id));
                state.cards = [...reordered, ...others];
                saveState();
            }
            evt.item.classList.add('settle-bounce');
            setTimeout(() => evt.item.classList.remove('settle-bounce'), 500);
